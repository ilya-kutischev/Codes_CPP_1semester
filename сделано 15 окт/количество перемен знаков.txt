//вводится н чисел с разными знаками, сосчитать количество перемен знаков не учитывая ноль
#include <iostream>
using namespace std;

int
FPeremen (int x) 
{
  int fl;
  
if (x > 0)
    fl = 1;
  
  else if (x < 0)
    fl = 0;
  
  else
    fl = 2;
  
 
return fl;

}


 
int
main ()
{
  
int i, n, fl1 = 2, chislo, fl, count = 0;
  
 
cout << "Vvedi n ";
  cin >> n;
  
for (i = 1; i <= n; i++)
    {
      
cout << "Vvedi chislo ";
      cin >> chislo;
      
fl = FPeremen (chislo);
      
if (fl == 2)
	{
	  cout << "Nol" << endl;
	}
      
      else if (fl == fl1)
	{
	  cout << "Ne izmenilsa" << endl;
	}
      
      else
	{
	  cout << "izmenilsa" << endl;
	  count++;
	}
      
fl1 = fl;
    
}
  
cout << "Vsego peremen znakov " << count - 1 << endl;
  
return 0;
  
system ("PAUSE");

}
